•%%% definition of the automaton directions
•%%% each Symbol that was read from the input changes the state.
•Branch(State,Symbol,Input) ::=
• { (State = "S0"), (Symbol = "a") , (Input =\= "") } , { State = S0 } | Extract ;
• { (State = "S0") ,(Symbol = "b") , (Input =\= "") } , { State = S1 } | Extract ;
• { (State = "S0") ,(Symbol = "a")  , (Input = "") } , Acceptor ;
• { (State = "S0") ,(Symbol = "b")  , (Input = "") } , DeadEnd ;
• { (State = "S1") , (Symbol = "a") , (Input =\= "") }, { State = S1 } | Extract ;
• { (State = "S1") , (Symbol = "b") , (Input =\= "") }, { State = S0 } | Extract ;
• { (State = "S1") ,(Symbol = "a") , (Input = "") }  , DeadEnd ;
• { (State = "S1") ,(Symbol = "b") , (Input = "") } , Acceptor ;
• { (Symbol =/= "b") }  , Error;
• { (Symbol =/= "a") }  , Error.
•
•% processes which stuck on dummy channel , dead end when a word was not accepted and acceptor when the word was accepted.
•Acceptor ::= dunny ? [] , true.
•DeadEnd ::= dunny ? [] , true.
•Error ::= dunny ? [] , true.
