RTK Kinase – Phosphorylation:
Option II
RTK_KINASE_SITE::=
  CROSS_PHOSPHORYLATE + FULL_PHOSPHORYLATE .
CROSS_PHOSPHORYLATE::=
  backbone_intra1 ? {cross_motif} , cross_motif ! {p_tyr} ,
     RTK_KINASE_SITE.
FULL_PHOSPHORYLATE::=
  backbone_intra3 ? [] , ACTIVE_FULL .
ACTIVE_FULL::=
  backbone_intra2 ? {cross_motif} , cross_motif ! {p_tyr} ,
     ACTIVE_FULL ;
 backbone_intra3 ? [] , RTK_KINASE_SITE .
