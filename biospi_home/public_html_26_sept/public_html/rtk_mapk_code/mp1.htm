-language([evaluate,psifcp,compound,colon]).
-export("MP1").
baserate(1).
global([raf_mp1,erk_mp1,mek_mp1]).



MP1:-
<< mp1,mp2,mp3,mp4 .
      MP1_FREE_MEK_BS ,
      << MP1_FREE_ERK_BS + MP1_FREE_RAF_BS >> .

MP1_FREE_RAF_BS:-
      raf_mp1 ! {mp1,mp2} | mp2 ? [] | << MP1_FREE_ERK_BS + MP1_FREE_RAF_BS >> .

MP1_FREE_ERK_BS:-
      erk_mp1 ! {mp3,mp2} | mp2 ? [] | << MP1_FREE_ERK_BS + MP1_FREE_RAF_BS >> .

MP1_FREE_MEK_BS:-
      mek_mp1 ! {mp1, mp3, mp4} | mp4 ? [] | MP1_FREE_MEK_BS
 
>>
