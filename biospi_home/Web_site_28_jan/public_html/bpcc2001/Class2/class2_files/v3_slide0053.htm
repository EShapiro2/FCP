O_Double_Bound(el)::=
     el ! [] , O ;
     el ? [] , O .
O_Bound1(el)::=
     el ! [] , O ;
     e ? {electron1}, O_Bound2(el,electron1) .

   electron1 ! [] , O_Bound1(el) ;
   el ! [] , O_Bound1(electron1) .
