-language([evaluate,psifcp,compound,colon]).
-export("RAS").
baserate(1).
global([switchI,switchII_gnef,switchII_gap,gdp,gtp]).


RAS(env):-

<< backbone, ras_switch1, ras_switch2 . INACTIVE_SWITCH_I , INACTIVE_SWITCH_II   .

INACTIVE_SWITCH_I:-
      backbone ! [] | ACTIVE_SWITCH_I .

ACTIVE_SWITCH_I:-
      switchI ! {env,ras_switch2} | ras_switch2 ! [] | ACTIVE_SWITCH_I ;
      backbone ! [] | INACTIVE_SWITCH_I .

INACTIVE_SWITCH_II:-
      switchII_gnef ! {env, ras_switch1} | ras_switch1 ? {nuc} |
            << nuc=?=gtp | backbone ? [] | ACTIVE_SWITCH_II;
               otherwise | INACTIVE_SWITCH_II >> .

ACTIVE_SWITCH_II:-
      switchII_gap ! {env, ras_switch1} | ras_switch1 ? {nuc} | 
            << nuc=?=gdp | backbone ? [] | INACTIVE_SWITCH_II;
               otherwise | ACTIVE_SWITCH_II >>   
 
>>
.
