-language([evaluate,psifcp,compound,colon]).
-export("LIGAND").
baserate(1).
global([ligand_bind,dummy]).


LIGAND:-

      <<ligand1,ligand2 . FREE_BD(ligand1,ligand2) , FREE_BD(ligand2,ligand1) .

      FREE_BD(lig1,lig2):-
          ligand_bind ! {lig1,lig2} | BOUND_BD .
     
      BOUND_BD:-
          dummy ? [] | true

      >>
